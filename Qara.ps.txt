<script>
    // دالة فتح وإغلاق القوائم المنسدلة
    function toggleAccordion(id, element) {
        var content = document.getElementById(id);
        if (content.style.display === "block") {
            content.style.display = "none";
            element.classList.remove("active");
        } else {
            content.style.display = "block";
            element.classList.add("active");
        }
    }

    // دالة لتعديل اسم رب الأسرة وعكسه في الترحيب علوياً
    function saveFatherName() {
        var newName = document.getElementById('input-father-name').value;
        document.getElementById('display-father-name').innerText = newName;
        alert('تم تحديث اسم رب الأسرة بنجاح!');
    }

    // إظهار نموذج إضافة ابن جديد
    function showAddForm() {
        var form = document.getElementById('add-child-form');
        if(form.style.display === "none") {
            form.style.display = "block";
            document.getElementById('toggle-add-btn').innerText = "إلغاء الإضافة";
        } else {
            form.style.display = "none";
            document.getElementById('toggle-add-btn').innerText = "إضافة ابن جديد +";
        }
    }

    // دالة إضافة ابن جديد ديناميكياً وحساب عمره
    function submitNewChild() {
        var name = document.getElementById('new-name').value;
        var gender = document.getElementById('new-gender').value;
        var idNum = document.getElementById('new-id').value;
        var birthDate = document.getElementById('new-date').value;

        if(!name || !idNum || !birthDate) {
            alert('الرجاء تعبئة كافة حقول الابن!');
            return;
        }

        // حساب العمر بشكل تقريبي بناء على تاريخ الميلاد
        var birthYear = new Date(birthDate).getFullYear();
        var currentYear = new Date().getFullYear();
        var age = currentYear - birthYear;

        var childId = 'child-' + Date.now(); // إنشاء ID فريد

        // بناء كود البطاقة الجديدة للابن
        var cardHTML = `
            <div class="child-card" id="${childId}">
                <div class="verified-badge">✓</div>
                <div class="child-name">${name}</div>
                <div class="child-gender">(${gender})</div>
                <div class="child-info">🪪 هوية: ${idNum}</div>
                <div class="child-info">📅 الميلاد: ${birthDate}</div>
                <div class="child-age">⏳ العمر الحالي: ${age} سنة</div>
                <div class="card-actions">
                    <button class="btn-edit" onclick="editChild('${childId}')">تعديل</button>
                    <button class="btn-delete" onclick="deleteChild('${childId}')">حذف</button>
                </div>
            </div>
        `;

        // إضافة البطاقة في الحاوية الرئيسية
        document.getElementById('children-container').insertAdjacentHTML('beforeend', cardHTML);

        // تنظيف الحقول وإخفاء النموذج
        document.getElementById('new-name').value = '';
        document.getElementById('new-id').value = '';
        document.getElementById('new-date').value = '';
        showAddForm();
    }

    // دالة حذف ابن من النظام
    function deleteChild(id) {
        if(confirm('هل أنت متأكد من حذف هذا الابن من النظام؟')) {
            document.getElementById(id).remove();
        }
    }

    // دالة تعديل بيانات ابن
    function editChild(id) {
        var card = document.getElementById(id);
        var currentName = card.querySelector('.child-name').innerText;
        var newName = prompt("قم بتعديل الاسم:", currentName);
        if (newName != null && newName !== "") {
            card.querySelector('.child-name').innerText = newName;
        }
    }
</script>
